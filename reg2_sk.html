LiteGraph.registerNodeType("basic/const", ConstantNode);
// where "basic/const" is the registered name of the Node, 
// you can refer to this node with this Name in the next step

const nodeConst1 = LiteGraph.createNode("basic/const");
// createNode creates a Node of type "basic/const"
nodeConst1.pos = [200, 200];
// pos determines the placement of the Node in the Graph
graph.add(nodeConst1);
// Add the newly created Node to the Graph
nodeConst1.setValue(4.5);
// Set a starting value of 4.5 to the Node

graph.start()
